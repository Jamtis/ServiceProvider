    ServiceProvider

    Example:

    import ServiceProvider = "node_modules/servingjs/build/ServiceProvider.js";
const manifest = {
    example_function(a,b) {
        return a + b;
    }
};
const options = {
    logging: false,
    isAuthorized() {
        return true; 
    }
};
const service_provider = new ServiceProvider(manifest, options);
const server = service_provider.startServer(8000);
// server.close();
